FROM php:8.2-apache

# 1. System-Abhängigkeiten installieren
# libonig-dev ist für mbstring wichtig, libxml2-dev für XML-Funktionen
RUN apt-get update && apt-get install -y \
    libicu-dev \
    libpng-dev \
    libonig-dev \
    libxml2-dev \
    zip \
    unzip \
    git \
    curl \
    && apt-get clean && rm -rf /var/lib/apt/lists/*

# 2. PHP Extensions installieren und aktivieren
# intl: Erforderlich für CI4 (Time, Locale, Validation)
# mbstring: Erforderlich für Textverarbeitung
# gd: Für Bildmanipulation
# mysqli/pdo_mysql: Für Datenbankverbindung
RUN docker-php-ext-configure intl \
    && docker-php-ext-install \
    intl \
    mbstring \
    gd \
    mysqli \
    pdo_mysql \
    xml \
    dom

# 3. Apache mod_rewrite aktivieren
RUN a2enmod rewrite

# 4. Apache Document Root auf /public setzen (CI4 Standard)
ENV APACHE_DOCUMENT_ROOT /var/www/html/public
RUN sed -ri -e 's!/var/www/html!${APACHE_DOCUMENT_ROOT}!g' /etc/apache2/sites-available/*.conf
RUN sed -ri -e 's!/var/www/html!${APACHE_DOCUMENT_ROOT}!g' /etc/apache2/apache2.conf /etc/apache2/conf-available/*.conf

# 5. Composer installieren
COPY --from=composer:latest /usr/bin/composer /usr/bin/composer

WORKDIR /var/www/html
